void ProcessSymbol(string symbol, IndicatorHandles &handles, datetime &lastBar, int magicNumber)
{
    // Execute once per M15 bar close
    datetime currentBarTime = iTime(symbol, PERIOD_M15, 0);
    if(currentBarTime == lastBar)
        return;
    lastBar = currentBarTime;
    
    // Check if max positions reached
    if(CountOpenPositions(magicNumber) > 0)
        return;  // Already have position on this symbol
    
    if(CountTotalOpenPositions() >= MaxConcurrentTrades)
        return;  // Max concurrent trades reached
    
    // === STEP 1: TIME WINDOW VALIDATION ===
    int timeWindow = GetTimeWindowTier();
    if(timeWindow == 0)
    {
        // Outside valid trading windows
        return;
    }
    
    // === STEP 2: TREND ALIGNMENT CHECK ===
    int trendDirection = GetTrendDirection(symbol, handles);
    if(trendDirection == 0)
    {
        // No clear trend - skip
        return;
    }
    
    // === STEP 3: VOLATILITY CONFIRMATION ===
    if(!CheckVolatilityExpansion(symbol, handles))
    {
        // ATR not expanding - avoid choppy market
        return;
    }
    
    // === STEP 4: ENTRY SIGNAL DETECTION ===
    int signal = DetectEntrySignal(symbol, trendDirection);
    
    if(signal == 1)  // BUY signal
    {
        double confidence = CalculateSignalConfidence(symbol, signal, timeWindow, trendDirection);
        if(confidence >= 15.0)  // Minimum confidence threshold
        {
            ExecuteTrade(symbol, ORDER_TYPE_BUY, handles, magicNumber);
        }
    }
    else if(signal == -1)  // SELL signal
    {
        double confidence = CalculateSignalConfidence(symbol, signal, timeWindow, trendDirection);

    if(CopyClose(symbol, PERIOD_M15, 0, bars, close) <= 0) return 0;
    if(CopyOpen(symbol, PERIOD_M15, 0, bars, open) <= 0) return 0;
    
    // Find recent high/low (excluding current bar)
    double recent_high = high[ArrayMaximum(high, 1, Breakout_Lookback)];
    double recent_low = low[ArrayMinimum(low, 1, Breakout_Lookback)];
    
    // === CHECK LONG SIGNAL ===
    if(allowedDirection == 1)  // Trend allows longs
    {
        // Breakout condition
        bool breakout = (close[0] > recent_high);
        
        // Momentum condition (higher lows over last 3 bars)
        bool momentum = true;
        for(int i = 1; i < MomentumBars; i++)
        {
            if(low[i] <= low[i+1])
                momentum = false;
        }
        
        // Bar strength (close in upper 30% of range)
        double range = high[0] - low[0];
        bool strongClose = (range > 0 && (close[0] - low[0]) / range > 0.7);
        
        if(breakout && momentum && strongClose)
        {
            Print("🔔 BUY SIGNAL on ", symbol, 
                  " | Breakout: ", DoubleToString(recent_high, _Digits),
                  " | Close: ", DoubleToString(close[0], _Digits));
            return 1;
        }
    }
    
    // === CHECK SHORT SIGNAL ===
    if(allowedDirection == -1)  // Trend allows shorts
    {
        // Breakdown condition
        bool breakdown = (close[0] < recent_low);
        
        // Momentum condition (lower highs over last 3 bars)
        bool momentum = true;
        for(int i = 1; i < MomentumBars; i++)
        {
            if(high[i] >= high[i+1])
                momentum = false;
        }
        
        // Bar strength (close in lower 30% of range)
        double range = high[0] - low[0];

                  " | Breakdown: ", DoubleToString(recent_low, _Digits),
                  " | Close: ", DoubleToString(close[0], _Digits));

    }

double CalculateLotSize(string symbol, double stopLossPoints)
{
    double accountBalance = AccountInfoDouble(ACCOUNT_BALANCE);
    double riskAmount = accountBalance * (RiskPercent / 100.0);  // $50 for $5000
    
    double tickValue = SymbolInfoDouble(symbol, SYMBOL_TRADE_TICK_VALUE);
    double tickSize = SymbolInfoDouble(symbol, SYMBOL_TRADE_TICK_SIZE);
    double pointValue = SymbolInfoDouble(symbol, SYMBOL_POINT);
    
    double minLot = SymbolInfoDouble(symbol, SYMBOL_VOLUME_MIN);
    double maxLot = SymbolInfoDouble(symbol, SYMBOL_VOLUME_MAX);
    double lotStep = SymbolInfoDouble(symbol, SYMBOL_VOLUME_STEP);
    
    // Calculate required lot size
    // Formula: Lot = Risk$ / (SL_points * PointValue * TickValue/TickSize)
    double lotSize = riskAmount / (stopLossPoints * pointValue * (tickValue / tickSize));
    
    // Normalize to lot step
    lotSize = MathFloor(lotSize / lotStep) * lotStep;
    
    // Apply limits
    if(lotSize < minLot)
    {
        Print("⚠ Calculated lot size ", lotSize, " below minimum ", minLot, 
              " for ", symbol, " - Using minimum");
        lotSize = minLot;
    }
    if(lotSize > maxLot)
    {
        Print("⚠ Calculated lot size ", lotSize, " above maximum ", maxLot,
              " for ", symbol, " - Using maximum");
        lotSize = maxLot;
    }
    
    Print("💰 Position Size: ", DoubleToString(lotSize, 2), " lots",
          " | Risk: $", DoubleToString(riskAmount, 2),
          " | SL: ", DoubleToString(stopLossPoints, 1), " points");
    
    return lotSize;
}

//| Execute Trade with 3-Position Scaling                             |
//+------------------------------------------------------------------+
void ExecuteTrade(string symbol, ENUM_ORDER_TYPE orderType, 
                  IndicatorHandles &handles, int magicNumber)
{
    double atr[];
    ArraySetAsSeries(atr, true);
    if(CopyBuffer(handles.atr_m15, 0, 0, 3, atr) <= 0)
    {
        Print("ERROR: Cannot read ATR for ", symbol);
        return;
    }
    
    double currentATR = atr[0];
    double point = SymbolInfoDouble(symbol, SYMBOL_POINT);
    
    // Calculate stop loss distance (2 × ATR)
    double slDistance = currentATR * StopLoss_ATR_Multiple;
    double slPoints = slDistance / point;
    
    // Calculate total lot size needed
    double totalLotSize = CalculateLotSize(symbol, slPoints);
    
    // Split into 3 positions
    double lot1 = MathFloor((totalLotSize * 0.33) * 100) / 100;  // 33%
    double lot2 = MathFloor((totalLotSize * 0.33) * 100) / 100;  // 33%
    double lot3 = totalLotSize - lot1 - lot2;                    // Remainder
    
    // Get current price
    double ask = SymbolInfoDouble(symbol, SYMBOL_ASK);
    double bid = SymbolInfoDouble(symbol, SYMBOL_BID);
    double price = (orderType == ORDER_TYPE_BUY) ? ask : bid;
    
    // Calculate stop loss and take profits
    double stopLoss, tp1, tp2, tp3;
    
    if(orderType == ORDER_TYPE_BUY)
    {
        stopLoss = ask - slDistance;
        tp1 = ask + (slDistance * TakeProfit1_RR);  // 1.5R
        tp2 = ask + (slDistance * TakeProfit2_RR);  // 3R
        tp3 = ask + (slDistance * TakeProfit3_RR);  // 5R
    }
    else  // SELL
    {
        stopLoss = bid + slDistance;
        tp1 = bid - (slDistance * TakeProfit1_RR);
        tp2 = bid - (slDistance * TakeProfit2_RR);
        tp3 = bid - (slDistance * TakeProfit3_RR);
    }
    
    // Normalize prices
    stopLoss = NormalizeDouble(stopLoss, _Digits);
    tp1 = NormalizeDouble(tp1, _Digits);
    tp2 = NormalizeDouble(tp2, _Digits);
    tp3 = NormalizeDouble(tp3, _Digits);
    
    Print("========================================");
    Print("EXECUTING TRADE ON ", symbol);
    Print("Direction: ", (orderType == ORDER_TYPE_BUY ? "BUY" : "SELL"));
    Print("Total Lots: ", totalLotSize, " (split: ", lot1, "+", lot2, "+", lot3, ")");
    Print("Entry: ", DoubleToString(price, _Digits));
    Print("Stop Loss: ", DoubleToString(stopLoss, _Digits), 
          " (", DoubleToString(slDistance/point, 0), " pts)");
    Print("TP1: ", DoubleToString(tp1, _Digits), " | ",
          "TP2: ", DoubleToString(tp2, _Digits), " | ",
          "TP3: ", DoubleToString(tp3, _Digits));
    Print("========================================");
    
    // Send orders
    SendOrder(symbol, orderType, lot1, stopLoss, tp1, magicNumber, "EA_Pos1");
    Sleep(100);  // Small delay between orders
    SendOrder(symbol, orderType, lot2, stopLoss, tp2, magicNumber, "EA_Pos2");
    Sleep(100);
    SendOrder(symbol, orderType, lot3, stopLoss, tp3, magicNumber, "EA_Pos3");
}

    else
    {

              " | Code: ", result.retcode,

        datetime openTime = (datetime)PositionGetInteger(POSITION_TIME);
        
        double currentPrice = (posType == POSITION_TYPE_BUY) ?
                             SymbolInfoDouble(symbol, SYMBOL_BID) :
                             SymbolInfoDouble(symbol, SYMBOL_ASK);
        
        // === BREAKEVEN STOP LOGIC ===
        // Move stop to breakeven when profit >= 1.5 × ATR
        double atrValue = GetCurrentATR(symbol);
        double profitPoints = MathAbs(currentPrice - openPrice) / SymbolInfoDouble(symbol, SYMBOL_POINT);
        double atrPoints = atrValue / SymbolInfoDouble(symbol, SYMBOL_POINT);
        
        if(profitPoints >= atrPoints * 1.5)
        {
            bool needsBreakeven = false;
            
            if(posType == POSITION_TYPE_BUY && currentSL < openPrice)
                needsBreakeven = true;
            else if(posType == POSITION_TYPE_SELL && currentSL > openPrice)
                needsBreakeven = true;
            
            if(needsBreakeven)
            {
                MoveStopToBreakeven(ticket, symbol, openPrice, currentTP);
            }
        }
        
        // === TRAILING STOP (For Position 3 only) ===
        string comment = PositionGetString(POSITION_COMMENT);
        if(StringFind(comment, "Pos3") >= 0)  // Position 3 uses trailing stop
        {
            ApplyTrailingStop(ticket, symbol, posType, currentPrice, 
                             currentSL, atrValue);
        }
        
        // === TIME-BASED EXIT ===
        int hoursOpen = (int)((TimeCurrent() - openTime) / 3600);
        if(hoursOpen >= MaxHoldTime_Hours)
        {
            Print("⏰ Time-based exit: ", symbol, " held for ",

void ApplyTrailingStop(ulong ticket, string symbol, ENUM_POSITION_TYPE posType,
                       double currentPrice, double currentSL, double atrValue)
{
    double trailDistance = atrValue * TrailingStop_ATR;
    double newSL = 0;
    
    if(posType == POSITION_TYPE_BUY)
    {
        newSL = currentPrice - trailDistance;
        
        // Only move SL up, never down
        if(newSL > currentSL)

            MqlTradeRequest request = {};
            MqlTradeResult result = {};

            request.position = ticket;
            request.symbol = symbol;
            request.sl = newSL;
            request.tp = PositionGetDouble(POSITION_TP);
            
            if(OrderSend(request, result))
            {
                Print("📈 Trailing stop adjusted UP | ", symbol,
                      " | New SL: ", DoubleToString(newSL, _Digits));
            }
        }
    }
    else  // SELL
    {
        newSL = currentPrice + trailDistance;
        
        // Only move SL down, never up
        if(newSL < currentSL || currentSL == 0)
        {
            newSL = NormalizeDouble(newSL, _Digits);

    }
}
